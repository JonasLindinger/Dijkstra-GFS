#!/bin/bash

input="GFS.mp4"
fps=60
numbers=(0 446 64 654 363 5 52 38 35 49 90 37 57 87 36 92 31 81 385 311 127 196 205 255 197 417 122 165 80 75 121 116 64 103 133 103 76 57 48 73 65 40 77 67 423 985 247 124 233 140 281 054 199 336 861 688 191 451 210 899 360 181 320 271 175 127 297 475 192)  # your list of numbers
count=1

start_frame=${numbers[0]}

for ((i=1; i<${#numbers[@]}; i++)); do
    length=${numbers[i]}
    end_frame=$((start_frame + length - 1))
    
    start_sec=$(echo "scale=3; $start_frame / $fps" | bc)
    end_sec=$(echo "scale=3; $end_frame / $fps" | bc)
    
    ffmpeg -i "$input" -ss "$start_sec" -to "$end_sec" -c copy "clip_$count.mp4"
    
    start_frame=$((end_frame + 1))
    ((count++))
done
